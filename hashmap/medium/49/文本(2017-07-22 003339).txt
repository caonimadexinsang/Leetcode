class Solution {public:    vector<vector<string>> groupAnagrams(vector<string>& strs) {        vector<vector<string>> result;        int length = strs.size();        if(length <= 0)            return result;	sort(strs.begin(),strs.end());	map<string, vector<string>> transmedium;	for (int i = 0; i < length; i++){		string t = strs[i];	    sort(t.begin(),t.end());		transmedium[t].push_back(strs[i]);	}	for (map<string, vector<string>>::iterator iter = transmedium.begin(); iter != transmedium.end(); iter++){		result.push_back(iter->second);	}        return result;    }};
